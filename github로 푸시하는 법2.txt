# 1. gh-pages 브랜치를 별도 디렉토리에 연결 (최초 1회만 필요)
git worktree add C:\wordchain\gh-pages gh-pages

# 2. main 브랜치에서 소스 코드 커밋 & 푸시
cd C:\wordchain
git add .
git commit -m "Update source code"
git push origin main

# 3. 빌드 실행 (main 브랜치에서)
npm run build

# 4. 빌드 결과물을 gh-pages 디렉토리로 복사
Copy-Item -Path C:\wordchain\dist\* -Destination C:\wordchain\gh-pages\ -Recurse -Force

# 5. gh-pages 디렉토리로 이동
cd C:\wordchain\gh-pages

# 6. 변경사항 커밋 & 푸시
git add .
git commit -m "Deploy new build"
git push origin gh-pages
